<script>
  var c = document.getElementById("canvas").getContext("2d");
  c.fillStyle="blue";
  c.fillRect(250, 25, 150, 150);

  c.beginPath();
  c.arc(280, 150, 100, Math.PI * 1/2, Math.PI * 3/2, true);
  c.lineWidth = 15;
  c.lineCap = 'round';
  c.strokeStyle = 'rgba(255, 127, 0, 0.5)';
  c.stroke();

  c.beginPath();
  c.bezierCurveTo(400, 20, 750, 280, 500, 200);
  c.lineWidth=5;
  c.strokeStyle="rgba(255,0,0, 0.8)";
  c.stroke();
</script>
